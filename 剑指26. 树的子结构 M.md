# 剑指26. 树的子结构 M



```js
/**
 * @param {TreeNode} A
 * @param {TreeNode} B
 * @return {boolean}
 */
var isSubStructure = function(A, B) {
    //按题设约定，A或B任一为空的边界情况下都为false
    if( !A || !B )
        return false;
    //满足三种情况之一返回true：A和B完全相同、A的左子树包含B、A的右子树包含B
    return recur( A, B ) || isSubStructure( A.left, B ) || isSubStructure( A.right, B);
    
};
//判断以A为根节点的树是否与B完全相同
var recur = function( A, B ){
    //当B为空为true
    if( !B )
        return true;
    //当B非空而A为空，或当前节点不同，为false
    if( !A || A.val !== B.val )
        return false;
    //仅当节点相同，递归判断左右子树
    return recur( A.left, B.left ) && recur( A.right, B.right );
}
```
