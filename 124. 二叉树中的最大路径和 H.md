# 124. 二叉树中的最大路径和 H

```js
/**
 * @param {TreeNode} root
 * @return {number}
 */

const maxPathSum = function(root) {

    let ans = root.val;

    const dfs = function( root ){
        
        if( !root )
            return 0;
        
        //对于叶子节点，尝试更新ans，然后直接返回其节点值作为该子树的收益
        if( !root.left && !root.right ){
            ans = Math.max( ans, root.val );
            return root.val;
        }
            
        //左右子树的收益如果为负，直接舍弃
        const gainL = Math.max( dfs( root.left ) , 0 );
        const gainR = Math.max( dfs( root.right ), 0 );

        //仅在本子树内的收益和
        const gainSelf = gainL + gainR + root.val;
        
        //尝试更新最大收益
        ans = Math.max( ans, gainSelf );
    
        //所能贡献给上层的收益
        const gainUp = Math.max( gainL, gainR ) + root.val;
        
        return gainUp;
    }

    //整个树的返回值也有可能成为最大和
    ans = Math.max( dfs( root ), ans );

    return ans;
};
```

Time: `O(N)`

Space: `O(Height)`
