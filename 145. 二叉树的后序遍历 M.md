# 145. 二叉树的后序遍历 M

## 一、迭代（一个栈）

```js
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
const postorderTraversal = function(root) {
    let res = [];
    if( !root )
        return res;
	//cur表示栈顶节点， pre表示上一个打印的节点（初始化为root以便最初的push）
    let cur = null, pre = root;
    let stk = [root];
    while( stk.length ){
        cur = stk[stk.length-1];
     
        if( cur.left && pre !== cur.left && pre !== cur.right ){
            stk.push( cur.left );
        }
        else if( cur.right && pre !== cur.right ){
            stk.push( cur.right );
        }
        else{
            pre = cur;
            res.push( cur.val );
            stk.pop();
        }

    }
    return res;
};
```



## 二、递归

```js
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
const postorderTraversal = function(root) {
    let res = [];
    
    const dfs = function( root ){
        if ( !root ) return;
        dfs( root.left );
        dfs( root.right );
        res.push( root.val );
    }
    dfs( root );
    return res;
};
```
