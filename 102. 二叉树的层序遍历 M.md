# 102. 二叉树的层序遍历 M

## 一、队列

```js
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
const levelOrder = function(root) {

    if( !root )
        return [];
    let res = [];

    let thisFloor = [root];
    
    while( thisFloor.length ){
        
        let thisRes = [];
        let thisFloorNodes = thisFloor.length;

        while( thisFloorNodes ){

            const node = thisFloor.shift();
            thisRes.push( node.val );

            if( node.left )
                thisFloor.push( node.left );
            if( node.right )
                thisFloor.push( node.right ); 
            
            thisFloorNodes --;
        }
        
        res.push( thisRes );
    }

    return res;
};
```

Time: `O(N)`

Space: `O(N)`

## 二、递归

```js
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
const levelOrder = function(root) {
    if( !root )
        return [];
    
    let res = [];

    const bfs = function( thisFloor, res ){

        let thisRes = [], nextFloor = [];
        while( thisFloor.length ){
            const node = thisFloor.shift();
            thisRes.push( node.val );

            if( node.left )
                nextFloor.push( node.left );
            if(node.right )
                nextFloor.push( node.right );
        }
        res.push( thisRes );

        if( nextFloor.length )
            bfs( nextFloor, res ); 

    }

    bfs( [root], res );

    return res;

};
```

Time: `O(N)`

Space: `O(N)`
