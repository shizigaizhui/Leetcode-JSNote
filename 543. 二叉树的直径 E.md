# 543. 二叉树的直径 E

```js
/**
 * @param {TreeNode} root
 * @return {number}
 */
const diameterOfBinaryTree = function(root) {

    let maxDiaNodes = 0;

    const depth = function( root ){
        
        if( !root ) 
            return 0;

        const leftDepth = depth( root.left );
        const rightDepth = depth( root.right );

        maxDiaNodes = Math.max( leftDepth + rightDepth + 1, maxDiaNodes );

        return Math.max( leftDepth, rightDepth ) + 1;
    }

    depth( root );
    //直径路径的节点数-1
    return maxDiaNodes - 1;
};
```

Time:`O(N)`

Space: `O(Height)`
