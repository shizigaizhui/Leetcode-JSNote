# 226. 翻转二叉树 E



```js
const invertTree = function(root) {

    if( !root || ( !root.left && !root.right ) )
        return root; 

    const leftSub = invertTree( root.left );
    const rightSub = invertTree( root.right );
    root.left = rightSub;
    root.right = leftSub;


    return root;
};
```
