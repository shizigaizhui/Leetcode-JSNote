# 96. 不同的二叉搜索树 M

## 动态规划

```js
/**
 * @param {number} n
 * @return {number}
 */
const numTrees = function(n) {

    // dp[i]表示i个递增节点的不同二叉搜索树的个数，边界值dp[0] = 1；
    // dp[i]从遍历dp[1]...dp[i-1]转移而来；
    let dp = new Array(n+1).fill(0);
    dp[0] = 1, dp[1] = 1;

    for(let i = 2; i <= n; ++i){
        for( let j = 1; j <= i; ++j){
            dp[i] += dp[j-1] * dp[i-j];
        }
    }

    return dp[n];
};
```
